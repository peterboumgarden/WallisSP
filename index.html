function unlock() {

  const k = document.getElementById('apikey').value.trim();

  if (!k.startsWith('sk-ant-') && !k.startsWith('sk-')) {

    document.getElementById('gate-err').style.display = 'block';

    document.getElementById('gate-err').textContent = 'Key should start with sk-ant-. Please check and try again.';

    return;

  }

  apiKey = k;

  document.getElementById('keygate').style.display = 'none';

  document.getElementById('app').classList.add('visible');

  renderChips();

}


document.getElementById('apikey').addEventListener('keydown', e => {

  if (e.key === 'Enter') unlock();

});


function setScope(el) {

  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));

  el.classList.add('active');

  currentScope = el.dataset.scope;

  const d = DATA[currentScope];

  document.getElementById('scope-title').textContent = d.title;

  document.getElementById('scope-sub').textContent = d.sub;

  const b = document.getElementById('scope-badge');

  b.textContent = d.badge;

  b.style.color = d.badgeColor;

  b.style.borderColor = d.badgeBorder;

  b.style.background = d.badgeBg;

  document.getElementById('qinput').placeholder = currentScope === 'all'

    ? 'Ask anything across all data layers...'

    : `Ask about ${d.title.toLowerCase()}...`;

  renderChips();

}


function renderChips() {

  const bar = document.getElementById('chips-bar');

  bar.innerHTML = '<span class="chips-lbl">Try →</span>';

  DATA[currentScope].chips.forEach(c => {

    const el = document.createElement('span');

    el.className = 'chip';

    el.textContent = c;

    el.onclick = () => { document.getElementById('qinput').value = c; ask(); };

    bar.appendChild(el);

  });

}


async function ask() {

  const q = document.getElementById('qinput').value.trim();

  if (!q || !apiKey) return;

  const btn = document.getElementById('askbtn');

  btn.disabled = true;


  const feed = document.getElementById('feed');

  const empty = document.getElementById('empty-state');

  if (empty) empty.remove();


  const thinking = document.createElement('div');

  thinking.className = 'thinking-card';

  thinking.innerHTML = '<div class="dots"><span></span><span></span><span></span></div><span>Searching knowledge base...</span>';

  feed.appendChild(thinking);

  feed.scrollTop = feed.scrollHeight;


  const d = DATA[currentScope];

  history.push({ role: 'user', content: q });


  try {

    const resp = await fetch('https://api.anthropic.com/v1/messages', {

      method: 'POST',

      headers: {

        'Content-Type': 'application/json',

        'x-api-key': apiKey,

        'anthropic-version': '2023-06-01',

        'anthropic-dangerous-direct-browser-access': 'true'

      },

      body: JSON.stringify({

        model: 'claude-sonnet-4-20250514',

        max_tokens: 1000,

        system: d.context,

        messages: history.slice(-10)

      })

    });


    if (!resp.ok) {

      const err = await resp.json().catch(() => ({}));

      throw new Error(err.error?.message || `API error ${resp.status}`);

    }


    const data = await resp.json();

    const answer = data.content?.filter(b => b.type === 'text').map(b => b.text).join('') || 'No response.';

    history.push({ role: 'assistant', content: answer });


    thinking.remove();


    const card = document.createElement('div');

    card.className = 'answer-card';

    const time = new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });

    card.innerHTML = `

      <div class="card-header">

        <div class="card-q" title="${q.replace(/"/g,'&quot;')}">${q}</div>

        <div style="display:flex;align-items:center;gap:8px;flex-shrink:0">

          <span class="scope-badge" style="font-size:10px;padding:2px 7px;color:${d.badgeColor};border-color:${d.badgeBorder};background:${d.badgeBg}">${d.badge}</span>

          <span class="card-time">${time}</span>

        </div>

      </div>

      <div class="card-body">${answer}</div>

      <div class="card-footer">

        <span class="source-pill">EnergyCoRP SharePoint</span>

        <span class="source-pill">${d.title}</span>

        ${currentScope === 'all' ? '<span class="source-pill">Cross-layer synthesis</span>' : ''}

      </div>`;

    feed.appendChild(card);

    feed.scrollTop = feed.scrollHeight;


  } catch(e) {

    thinking.remove();

    const err = document.createElement('div');

    err.className = 'answer-card';

    err.innerHTML = `<div class="card-body" style="color:var(--coral)">Error: ${e.message}. Check your API key and try again.</div>`;

    feed.appendChild(err);

    history.pop();

  }


  document.getElementById('qinput').value = '';

  btn.disabled = false;

  document.getElementById('qinput').focus();

}
